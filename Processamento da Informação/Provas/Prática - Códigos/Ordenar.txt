import java.util.Scanner;
import java.util.Arrays;

public class EP7_3 {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int t;
        System.out.print("Digite tamanho: ");
        t = input.nextInt();
        int v[] = new int[t];
        //vetor
        for (int i = 0; i < t; i++) {
            System.out.print("valor: ");
            v[i] = input.nextInt();
        }
        Arrays.sort(v);
        for (int h: v){
            System.out.print(h+" ");
        }
    }
}
